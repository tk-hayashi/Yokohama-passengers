def forecast(model, x, n, span):
    model.reset_states()
    Y = []
    x_tmp = x
    for i in range(n):
        x_tmp = x_tmp.reshape(1, span, 1)
        y = model.predict(x_tmp)
        Y.append(y)

        x_tmp = x_tmp.reshape(span, 1)
        x_tmp = np.vstack((x_tmp, y))
        x_tmp = np.delete(x_tmp, 0, 0)
        
    model.reset_states()
    return Y
